+SYSCALL_DEFINE2(sh_task_info,int, pid, char*,filename) {
+
+  char path[100],string[200];
+  loff_t set = 0;
+  mm_segment_t old_fs;
+  struct file *file_ptr=NULL;
+  struct task_struct *proces = NULL;
+  struct pid *pid_struct;
+  int result=0;
+  int output;
+  copy_from_user(path,filename,strnlen_user(filename,100));
+  
+  
+  if(pid>32768 || pid<=0)
+  {
+    return -EINVAL;
+  }
+  //proces=find_task_by_pid(pid);
+  
+  pid_struct = find_get_pid(pid);
+  proces = pid_task(pid_struct,PIDTYPE_PID);
+  if(proces==NULL)
+  {
+    return -EINVAL;
+  }
+
+  output=sprintf(string, "Process Details ->\nProcess: %s\nPID_Number: %ld\nProcess State: %ld\nStatic Priority: %ld\nNormal Priority: %ld\n", proces->comm, (long)task_pid_nr(proces), (long)proces->state, (long)proces->static_prio, (long)proces->normal_prio);
+
+  printk(string);
+  
+
+  old_fs = get_fs();
+  set_fs(KERNEL_DS);
+
+
+  file_ptr = filp_open(path, O_WRONLY | O_CREAT, 0644);
+  
+  if(IS_ERR(file_ptr)) {
+    return -PTR_ERR(file_ptr);
+  }
+
+  vfs_write(file_ptr, string, strlen(string), &set);
+  vfs_fsync(file_ptr, 0);
+  filp_close(file_ptr, NULL);
+
+  set_fs(old_fs);
+  return result;
+ }
